## useMoveToError

```jsx
import { useFormikContext } from 'formik';
import _ from 'lodash';
import { useEffect } from 'react';

function useMoveToError() {
  const formik = useFormikContext();

  useEffect(() => {
    const firstErrorField = _.keys(formik.errors).at(0);
    const el = document.querySelector(`input[name="${firstErrorField}"]`);
    (el?.parentElement ?? el)?.scrollIntoView({ behavior: 'smooth' });
  }, [formik?.isSubmitting, formik.errors]);

  return null;
}

export default useMoveToError;
```

## usage

```jsx
useMoveToError();
```
